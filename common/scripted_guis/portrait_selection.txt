	portrait_gui = {
		context_type = player_context
		window_name = "portrait_gui_container"
		parent_window_token = countrypoliticsview

		visible = {
            is_ai = no
		}

		effects = {
			portrait_gui_button_click = {
				set_temp_variable = { show = 1 }
				if = {
					limit = {
						has_country_leader = {
							character = GER_adolf_hitler
							ruling_only = yes
						}
					}
					subtract_from_temp_variable = { show = show_portrait_gui_hitler }
					set_variable = { show_portrait_gui_hitler = show }
					else_if = {
						limit = {
							has_country_leader = {
								character = SOV_iosif_stalin
								ruling_only = yes
							}
						}
						subtract_from_temp_variable = { show = show_portrait_gui_stalin }
						set_variable = { show_portrait_gui_stalin = show }
					}
					else_if = {
						limit = {
							has_country_leader = {
								character = SHX_yan_xishan
								ruling_only = yes
							}
						}
						subtract_from_temp_variable = { show = show_portrait_gui_xishan }
						set_variable = { show_portrait_gui_xishan = show }
					}
					else = {
						subtract_from_temp_variable = { show = show_portrait_gui_not_added }
						set_variable = { show_portrait_gui_not_added = show }
					}
				}
			}
		}
	}
